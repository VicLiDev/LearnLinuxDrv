### 11.2 添加一个新的命令

```c
// 1. 定义处理函数
COMMAND_HANDLER(handle_my_command)
{
    if (CMD_ARGC < 1) {
        command_print(CMD, "usage: my_command <arg>");
        return ERROR_COMMAND_SYNTAX_ERROR;
    }

    uint32_t val;
    COMMAND_PARSE_NUMBER(u32, CMD_ARGV[0], val);

    command_print(CMD, "got value: 0x%08x", val);
    return ERROR_OK;
}

// 2. 注册命令
static const struct command_registration my_handlers[] = {
    {
        .name = "my_command",
        .handler = handle_my_command,
        .mode = COMMAND_EXEC,
        .help = "my custom command",
        .usage = "<value>",
    },
    COMMAND_REGISTRATION_DONE
};

// 3. 在模块初始化时注册
int my_module_init(struct command_context *cmd_ctx)
{
    return register_commands(cmd_ctx, NULL, my_handlers);
}
```
